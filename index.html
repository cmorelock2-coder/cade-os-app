async function fetchXRP() {

  try {

    const r = await fetch('https://api.coinbase.com/v2/prices/XRP-USD/spot');

    const d = await r.json();

    const p = parseFloat(d.data.amount);

    const val = p * XRP_HELD;

    const ret = val - XRP_BASIS;

    const pct = ((ret/XRP_BASIS)*100).toFixed(1);

    const up = ret >= 0;

    const fmt = n => '$'+Math.abs(Math.round(n)).toLocaleString();

    document.getElementById('xrp-price').textContent = '$'+p.toFixed(4);

    document.getElementById('xrp-val').textContent = fmt(val);

    document.getElementById('xrp-val').className = 'xsv '+(up?'up':'dn');

    document.getElementById('xrp-ret').textContent = (up?'+':'-')+fmt(ret);

    document.getElementById('xrp-ret').className = 'xsv '+(up?'up':'dn');

    document.getElementById('xrp-pct').textContent = (up?'+':'')+pct+'%';

    document.getElementById('xrp-pct').className = 'xsv '+(up?'up':'dn');

    document.getElementById('xrp-chg').textContent = (up?'+':'')+pct+'%';

    document.getElementById('xrp-chg').className = 'chg '+(up?'chg-u':'chg-d');

    document.getElementById('dash-xrp').textContent = fmt(val);

    document.getElementById('dash-xrp-s').textContent = (up?'+':'')+pct+'% all-time';

    document.getElementById('dash-xrp-s').className = 'ks '+(up?'up':'dn');

    const pts = [0.48,0.51,0.50,0.55,0.58,0.54,p];

    const mx = Math.max(...pts);

    document.getElementById('spark').innerHTML = pts.map((v,i)=>

      `<div class="sb${i===pts.length-1?' pk':''}" style="height:${Math.round((v/mx)*100)}%"></div>`

    ).join('');

  } catch(e) { console.log('XRP fetch error',e); }

}

async function sendMsg() {

  const inp = document.getElementById('chat-in');

  const txt = inp.value.trim();

  if(!txt) return;

  inp.value = '';

  const msgs = document.getElementById('chat-msgs');

  const t = ()=>new Date().toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'});

  msgs.innerHTML += `<div class="msg u"><div class="bubble">${txt}</div><div class="msg-t">${t()}</div></div>`;

  msgs.innerHTML += `<div class="msg b" id="typ"><div class="dots"><div class="dot"></div><div class="dot"></div><div class="dot"></div></div></div>`;

  msgs.scrollTop = msgs.scrollHeight;

  history.push({role:'user',content:txt});

  try {

    const res = await fetch('https://api.anthropic.com/v1/messages',{

      method:'POST',headers:{'Content-Type':'application/json'},

      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,system:SYS,messages:history})

    });

    const data = await res.json();

    const reply = data.content?.find(b=>b.type==='text')?.text || 'Something went wrong.';

    history.push({role:'assistant',content:reply});

    document.getElementById('typ')?.remove();

    msgs.innerHTML += `<div class="msg b"><div class="bubble">${reply.replace(/\n/g,'<br>')}</div><div class="msg-t">${t()}</div></div>`;

    msgs.scrollTop = msgs.scrollHeight;

  } catch(e) {

    document.getElementById('typ')?.remove();

    msgs.innerHTML += `<div class="msg b"><div class="bubble" style="color:var(--r)">Couldn't reach agent.</div></div>`;

  }

}
